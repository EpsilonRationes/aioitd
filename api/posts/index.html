    TYPE: str DEFAULT: ''

    attachment_ids

    TYPE: list[UUID] | None DEFAULT: None

    TYPE: UUID DEFAULT: None

    multiple_choice

    TYPE: bool DEFAULT: False

    question

    options

    TYPE: list[str] | None DEFAULT: None

    spans

    ValidationError

    ValidationError

    2 <= len(options) <= 10
    ValidationError

    1 <= len(options[i]) <= 32
